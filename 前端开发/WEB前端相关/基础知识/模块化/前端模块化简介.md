# AMD
是 require.js 的规范。
```js
// main.js
require(['moduleA'], function(a){  // 使用 require 加载模块
　　console.log(a);
});

//moduleA.js
define(function(){  // 使用 define 定义模块
    return 1;
})
```

# CMD
是 sea.js 的规范。
```js
// main.js
define(function(require, exports, module){
    document.onclick = function(){
        var a = require('js/a');  // 使用 require 加载模块
        a.test();
    }    
});

// a.js
define(function(require, exports, module){  // 使用 define 定义模块
    console.log('a');
    exports.test = function(){
        console.log('a.test');
    }
})
```

# CommonJS
node.js 使用的模块规范。
```js
// example.js
var x = 5;
var addX = function (value) {
    return value + x;
};
module.exports.x = x;  // 使用 module.exports 导出模块
module.exports.addX = addX;
```
```js
// main.js
var example = require('./example.js');  // 使用 require 导入模块

console.log(example.x); // 5
console.log(example.addX(1)); // 6
```

# ES Module
ES6 发布时制定的模块化规范，主要用于前端领域。
```js
// example.js
var x = 5;
var addX = function (value) {
    return value + x;
};
export { x, addX };  // 使用exports 导出模块
```
```js
// main.js
import example from './example.js';  // 使用 import 导入模块

console.log(example.x); // 5
console.log(example.addX(1)); // 6
```

# UMD
UMD 并不算是模块规范，它是为模块化做兼容性处理的，判断使用者所处的模块环境来实现对应模块规范的导出。
```js
(function (root, factory) {
    if (typeof define === 'function' && define.amd) {
        // AMD. Register as an anonymous module.
        define(['b'], factory);
    } else if (typeof module === 'object' && module.exports) {
        // Node. Does not work with strict CommonJS, but
        // only CommonJS-like environments that support module.exports,
        // like Node.
        module.exports = factory(require('b'));
    } else {
        // Browser globals (root is window)
        root.returnExports = factory(root.b);
    }
}(this, function (b) {
    //use b in some fashion.

    // Just return a value to define the module export.
    // This example returns an object, but the module
    // can return a function as the exported value.
    return {}
}));
```

在使用 webpack 打包一个公共库时，通常都是打包成 UMD，这样才有更好的兼容性。
